91  /**
92  * this method takes an ItemType as a parameter and uses it to iterate through the NameValueListType[]
93  * contained within it adding the appropriate names and values to the this.itemSpecifics_M map.
94  * Before adding the 'name' field to the key of the map it removes the spaces in the name, so
95  * to access 'Manufacturer Part Number' you would request 'ManufacturerPartNumber' etc.
96  * @param item the ItemType returned from the API call.
97  */
98  private void populateMap(ItemType item)
99  {
100  for (NameValueListType specific : item.getItemSpecifics().getNameValueList())
101  {
102  String value = Arrays.toString(specific.getValue());
103  value = value.substring(1, value.length()-1);
104  this.itemSpecifics_M.put(specific.getName().replace(" ", ""), value);
105  }
106  }
