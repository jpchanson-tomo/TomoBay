24  * This class defines a service scheduler, it runs services intervalicaly, with the interval
25  * being defined from the previous termination of that service thread. Contains a threadpool
26  * which should be changed to reflect the number of services stored in the queue, for small
27  * numbers of services; and set to a sensible value for larger numbers of services.
28  * @author Jan P.C. Hanson
29  *
30  */
31 public class ServiceScheduler

33  /**Scheduled thread pool**/

35  /**List of Scheduled futures representing service threads**/
36  private List<ScheduledFuture<?>> services_M;

38  /**
39  * constructor creates a thread pool with the number of threads specified in the arguments.
40  * @param noOfThreads the number of threads to populate the pool with.
41  */
42  public ServiceScheduler(int noOfThreads)
43  {
44  this.serviceScheduler_M = new ScheduledThreadPoolExecutor(noOfThreads);
45  this.services_M = new ArrayList<ScheduledFuture<?>>();
46  }
47 
48  /**
49  * add a service to the list of services to run at the periods specified in the method
50  * parameters. After the initial 'delay' (in minutes) the 'service' executes, once execution
51  * has finished, the service will be executed again after the number of minutes specified in
52  * the 'rate' parameter.
53  * @param service the service to add
54  * @param delay the initial delay before running the service
55  * @param rate the rate at which to repeat the service.
56  */
57  public void add(AbstractService service, long delay, long rate)
58  {
59  this.services_M.add(this.serviceScheduler_M.scheduleWithFixedDelay(service, delay, rate, TimeUnit.MINUTES));
60  }
61 
62  /**
63  * starts the service.
64  */
65  public void start()
66  {
67  try

69  for(int i = 0 ; i < this.services_M.size() ; ++i)

71  this.services_M.get(i).get();
72  }
73  }
74  catch(InterruptedException | ExecutionException e)
75  {e.printStackTrace();}

78  }
