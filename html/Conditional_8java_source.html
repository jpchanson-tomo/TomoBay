18 /**
19  * This class attempts to represent conditional statements in an object oriented manner. A Condition
20  * object is evaluated and depending on whether the Condition evaluates to true or false the appropriate
21  * code in the Result object is invoked (using double dispatch).
22  *
23  * Two evaluate methods are defined here: evaluate() and evaluate(Condition condition, Result<T> result);
24  * which is to be used depends on how you wish to use these constructs. The parameterless evaluate()
25  * method is provided in the case where you need to define a particular start condition and chain
26  * conditional statements, much like a nested if-then-else statement. The evaluate(Condition condition, Result<T> result)
27  * is provided in the case where a simple if-else statement is needed.
28  *
29  * @see tomoBay.model.dataTypes.conditionalStatement
30  *
31  * @author Jan P.C. Hanson
32  *
33  */
34 @SuppressWarnings("unused")
35 public abstract class Conditional<T>
36 {

38  * default ctor
39  */
40  public Conditional()
41  {super();}

43  /**
44  * Evaluate the start condition inherent to this conditional (if it exists).
45  * @return
46  */
47  public T evaluate()
48  {return this.startCondition().evaluate().dispatch(this.startResult());}

50  /**
51  * evaluate this conditional based on the Condition and Result objects passed as parameters to
52  * this method
53  * @param condition the Condition object that should be evaluated.
54  * @param result the Result object that contains the behaviour to invoke for this conditional.
55  * @return T as defined when instantiating this Conditional object.
56  */
57  public T evaluate(Condition condition, Result<T> result)
58  {return condition.evaluate().dispatch(result);}
59 
60  /**

64  */

72  */
73  public abstract Result<T> startResult();
