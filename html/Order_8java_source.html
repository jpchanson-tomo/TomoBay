32  * This class represents an eBay order, and the relevant information associated with it. Once
33  * instantiated this object will contain a Buyer object and an array of Transaction objects, and while
34  * this Order object will contain general information relevant to this eBay order, in order to gain
35  * more in depth details of the order i.e. buyer information or detailed information about the items
36  * that have been sold, it is necessary to query these contained objects.
37  * @author Jan P.C. Hanson
38  *
39  */

157  /**

173  /**

177  * @return Transaction[] that can be used to populate the final Transaction[] member variable.
178  */
179  private static final Transaction[] getTransactionList(String orderID)
180  {
181  HeteroFieldContainer params = new HeteroFieldContainer();
182  params.add(OrdersTable.ORDER_ID, orderID);
183  List<HeteroFieldContainer> query = SelectQueryInvoker.execute(
184  SelectQueryTypeParams.SELECT_TRANSACTION_BY_ORDERID,
185  params
186  );
187  Transaction[] result = new Transaction[query.size()];
188 
189  for(int i=0 ; i < query.size() ; ++i)
190  {result[i]=new Transaction(query.get(i).get(TransactionsTable.TRANSACTION_ID, ClassRef.LONG));}
191 
192  return result;
193  }
