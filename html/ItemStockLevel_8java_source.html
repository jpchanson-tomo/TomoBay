62  /**
63  * Takes the list of items generated in this.assembleItemsList and checks a PartList
64  * associated with each item against Stock held in winstock. For every item with enough
65  * stock level it adds a true value to a Set<Boolean> i.e. if everything is in stock the
66  * set will only contain 1 element = true, if everything is out of stock the set will
67  * contain only 1 element = false, but if some items are in stock and others arn't it will
68  * contain 2 elements true and false.
69  * @param itemsVar the list of items to check off against Stock levels
70  * @return Set<Boolean> in stock - size=1 value={true}; out of stock - size=1 values={false};
71  * partial stock - size = 2 values = {true,false}
72  */
73  private static Set<Boolean> checkStockLevelOfItems(List<HeteroFieldContainer> itemsVar)
74  {
75  final Set<Boolean> itemStatus = new THashSet<Boolean>();
76  for (HeteroFieldContainer item : itemsVar)
77  {
78  final Stock stockAvailable = new Stock();
79  final PartList partlist = new PartList(item.get(ItemsTable.PART_NO, ClassRef.STRING));
80 
81  for (int i = 0 ; i < partlist.size() ; ++i)
82  {
83  final int available = stockAvailable.requestStockLevel(partlist.getPartNumber(i),
84  BrandToCode.convertToWinstockInt(item.get(ItemsTable.BRAND, ClassRef.STRING))+"");
85 
86  final int required = partlist.getPartQty(i);
87 
88  boolean status = available >= required ? true : false;
89  if(available == Stock.ERROR) {return null;}
90 
91  itemStatus.add(status);
92  }
93  }
94  return itemStatus;
95  }
