18 /**
19  * This class represents an algorithm for adjusting the prices of an AbstractSalesDayBookLine
20  * so that the calculated total (i.e. the amount that is actually invoiced) is as close as
21  * physically possible to the listing price on ebay. This is necessary as the ebay based order
22  * price is inclusive of VAT however the number that I need to provide the winstock invoiceing
23  * system is inclusive of VAT. This is coupled with the fact that when listing items, the humans
24  * responsible do not always figure VAT in when deciding on a price.
25  *
26  * This class is final and ALL methods in this class are either protected or
27  * private so that this class cannot be subclassed and the methods in this class cannot be accessed
28  * outside of this package....(package scope is silly)
29  * @author Jan P.C. Hanson
30  *
31  */

42  /**
43  * calculate the adjustments necessary and apply them to the AbstractSalesDayBookLine provided
44  * as an argument
45  * @param line AbstractSalesDayBookLine to adjust
46  */
47  protected void adjust(AbstractSalesDayBookLine line)
48  {
49  int noOfItems=this.noOfItems(line);
50  int smallestQty = this.smallestQty(line);
51  int[] testPrices = this.convertListToArray(noOfItems, line);
52  int initialDiscrepancy = this.fitness(line, this.calculateTotal(testPrices,line));
53  int adjustAmount = this.adjustmentAmt(initialDiscrepancy);
54  int fitness = this.fitness(line, initialDiscrepancy);
55 
56  for(int i = 0 ; i < ((initialDiscrepancy)/smallestQty) ; ++i)
57  {
58  int bestChoice = PriceAdjustment.PLACEHOLDER;
59  for(int j = 0 ; j < noOfItems ; ++j)
60  {
61  testPrices[j] = testPrices[j] + adjustAmount;
62  int tmpFit = this.fitness(line, this.calculateTotal(testPrices,line));
63  testPrices[j] = testPrices[j] - adjustAmount;
64  if(tmpFit < fitness){bestChoice = j;fitness=tmpFit;}
65  }
66  if(bestChoice!=PriceAdjustment.PLACEHOLDER) {testPrices[bestChoice] += adjustAmount;}
67  }
68  this.performAdjustement(line, testPrices);
69  }
