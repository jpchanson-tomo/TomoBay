70  public void start(long rateInMins)

72  while(this.services_M.size() > 0)
73  {
74  try
75  {
76  for(int i = 0 ; i < this.services_M.size() ; ++i)

78  log.warn(this.serviceScheduler_M.submit(this.services_M.get(i)).get().toString());
79  System.runFinalization();
80  }

82  }
83  catch(InterruptedException | ExecutionException e)
84  {log.error("Service Error "+StackTraceToString.toString(e));}
85  }
86  serviceScheduler_M.shutdown();
