41  * This AbstractPresenterAction provides functionality to invoice a number of orders based on the
42  * order numbers provided as the data string to the execute method. This string should be a comma
43  * separated string of valid order numbers i.e. real order numbers that are less than 90 days old.

45  *
46  */

61  /* (non-Javadoc)
62  * @see tomoBay.presenters.presenterActions.AbstractPresenterAction#execute(java.lang.String)
63  */
64  @Override
65  public String execute(String data)
66  {
67 
68  String[] orderIDs = this.splitOrderNoString(data);
69  String result="";
70  for(String orderId : orderIDs)
71  {
72  try
73  {
74  AbstractSalesDayBookLine invoice = SalesDayBookLineFactory.make(SalesDayBookLineType.INVOICE, new Order(orderId));
75  if(invoice.invoiceNumber()==0 && TimeStampFunctions.olderThan(30, invoice.orderInfo().createdTime())==false)
76  {
77  DualList<String, PayloadType> winstockInv = this.formatAsDualList(invoice);
78  AbstractWinstockCommandResponse res;
79  res = WinstockCommandInvoker.execute(WinstockCommandInvoker.WinstockCommandTypes.PutInvoice, winstockInv);
80  result+=this.formatMailResult(res.getRecieved()[0], res.getRecieved()[1], invoice);
81  this.updateDB(res.getRecieved()[0], orderId);
82  }
83  else {result+="("+invoice.orderInfo().salesRecNo()+" already Invoiced or too old"+")";}
84  }
85  catch (UnknownHostException e){e.printStackTrace(); result+= "(UnKnownHost Error with "+orderId+"),";}
86  catch (IOException e){e.printStackTrace(); result+= "(IO Error with "+orderId+"),";}
87  catch (PayloadException e){e.printStackTrace(); result+= "(Payload Error with "+orderId+"),";}
88  catch (RuntimeException e) {e.printStackTrace(); result+= "(Runtime Error with "+orderId+"),";}
89  }
90  log.warn(result+"-----INVOICED");
91  this.emailResults(result);
92  return InvoiceOrders.EMAILHEADER + result + InvoiceOrders.EMAILFOOTER;
93  }
