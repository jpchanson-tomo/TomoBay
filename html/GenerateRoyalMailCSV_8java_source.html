94  /**
95  * takes the input string and makes it accessible as an array, the public static finals define the
96  * indices that the data is stored at.
97  * @param input String of the form
98  * "Account-SalesRecNo-Service-Service Enhancement-format|Account-SalesRecNo-Service-Service Enhancement-format|....."
99  * @return String[][] containing the intial data
100  *
101  */
102  private String[][] processInput(String input)
103  {
104  //splits the String into rows
105  String[] halfProcessed = input.replaceAll(" ", "").split("[|]");
106 
107  String[][] fullyProcessed = new String[halfProcessed.length][];
108  //splits each String in halfProcessed into an array
109  for(int i = 0 ; i < halfProcessed.length ; ++i)
110  {
111  fullyProcessed[i] = halfProcessed[i].replace(' ', '\u0000').split("-");
112  }
113  return fullyProcessed;
114  }

116  /**
117  * formats the initial data passed in to the execute method by the frontend and puts data into the
118  * relevant columns for the CSV.
119  * @param initialData the String containing the input data
120  * @return String[][] containing as many elements as there are entries inputted into the frontend
121  * each of these elements is an array containing 13 data points with the data taken from the input
122  * put in the right places.
123  */
124  private String[][] formatInitialData(String[][] initialData)
125  {
126  String[][] output = new String[initialData.length][csvGen_M.size()];
127  for(int i = 0 ; i < initialData.length ; ++i)
128  {
129  output[i][RoyalMailCSVType.SERVICE] = initialData[i][GenerateRoyalMailCSV.SERVICE];
130  output[i][RoyalMailCSVType.SERVICE_ENHANCEMENT] = initialData[i][GenerateRoyalMailCSV.SERVICE_ENHANCEMENT];
131  output[i][RoyalMailCSVType.SERVICE_FORMAT] = initialData[i][GenerateRoyalMailCSV.FORMAT];
132  output[i][RoyalMailCSVType.SERVICE_REF] = "SR1";
133  output[i][RoyalMailCSVType.COUNTRY_CODE] = "GB";
134  output[i][RoyalMailCSVType.ITEMS] = "1";
135  }
136  return output;
137  }

140  /**
141  * grabs the rest of the data necessary for the RoyalMailCSVGenerator to create a CSV file
142  * @param inputData the processed input data that acts as a key to grab the rest of the data from
143  * the database.
144  * @param initialData String[][] containing the intial data passed in from the frontend
145  * @return String[][] containing the data passed in with DB data inserted into relevant columns
146  */
147  private String[][] grabAndProcessDBData(String[][] inputData, String[][] initialData)
148  {
149  HeteroFieldContainer dbaseContainer = new HeteroFieldContainer();
150 
151  for(int i = 0 ; i < inputData.length ; ++i)
152  {
153 // prepare query and grab data from database
154  dbaseContainer.clear();
155  dbaseContainer.add(AccountsTable.ACCOUNT_NAME, initialData[i][GenerateRoyalMailCSV.ACCOUNT]);
156  dbaseContainer.add(OrdersTable.SALES_REC_NO, Integer.parseInt(initialData[i][GenerateRoyalMailCSV.SALES_REC_NO]));
157 
158  final List<HeteroFieldContainer> result =
159  SelectQueryInvoker.execute(SelectQueryTypeParams.SELECT_EBAY_BUYER_BY_ACCOUNT_AND_SRN, dbaseContainer);
160 
161  //format data
162  if(result.size() != 0)
163  {
164  inputData[i][RoyalMailCSVType.RECIPIENT] = result.get(0).get(BuyerTable.NAME, ClassRef.STRING);
165  inputData[i][RoyalMailCSVType.ADDRESS_LINE_1] = result.get(0).get(BuyerTable.STREET1, ClassRef.STRING);
166  inputData[i][RoyalMailCSVType.ADDRESS_LINE_2] = result.get(0).get(BuyerTable.STREET2, ClassRef.STRING);
167  inputData[i][RoyalMailCSVType.POSTCODE] = result.get(0).get(BuyerTable.POSTCODE, ClassRef.STRING).toUpperCase();
168  inputData[i][RoyalMailCSVType.POST_TOWN] = result.get(0).get(BuyerTable.CITY, ClassRef.STRING);
169  inputData[i][RoyalMailCSVType.REFERENCE]
170  = result.get(0).get(OrdersTable.INVOICED, ClassRef.INTEGER).toString();
171  }
172  else
173  {
174  throw new PresenterActionException(
175  "invalid sales record number or account: ["
176  +initialData[i][GenerateRoyalMailCSV.SALES_REC_NO]
177  +":"+initialData[i][GenerateRoyalMailCSV.ACCOUNT]+"]"
178  );
179  }
180  }
181  return inputData;
182  }

184  /**
185  * queries winstock for the weight associated with the invoice number provided as an argument.
186  * @param inputData the String[][] to insert the weight into.
187  * @param initialData String[][] containing the initial data passed in from the frontend
188  * @return String[][] containing the data passed in with the weight inserted into the appropriate
189  * column.
190  */
191  private String[][] getInvoiceWeight(String[][] inputData, String[][] initialData)
192  {
193  Stock winstock = new Stock();
194  for(int i = 0 ; i < inputData.length ; ++i)
195  {
196  final int invoiceNo = Integer.parseInt(inputData[i][RoyalMailCSVType.REFERENCE]);
197  final int brandCode = BrandToCode.convertToWinstockInt(initialData[i][GenerateRoyalMailCSV.ACCOUNT]);
198  final String weight = String.valueOf(winstock.requestInvoiceWeight(invoiceNo, brandCode));
199  if(weight.equals("0")) {inputData[i][RoyalMailCSVType.WEIGHT]= "1";}
200  else {inputData[i][RoyalMailCSVType.WEIGHT]= weight;}
201  }
202  return inputData;
203  }

205  /**
206  * adds the account letter to the front of the invoice number in the reference field of the CSV
207  * @param inputData the data after all extra information has been inserted
208  * @param initialData the initial data taken from the frontend
209  * @return String[][] completely formatted data ready to pass to the createCSV method.
210  */
211  private String[][] formatReferenceFields(String[][] inputData, String[][] initialData)
212  {
213  for(int i = 0 ; i < inputData.length ; ++i)
214  {
215  inputData[i][RoyalMailCSVType.REFERENCE] = BrandToCode.convert(initialData[i][GenerateRoyalMailCSV.ACCOUNT])
216  + inputData[i][RoyalMailCSVType.REFERENCE];
217  }
218  return inputData;
219  }
