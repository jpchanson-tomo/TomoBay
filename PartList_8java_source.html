93  /**
94  * orchestrates the conversion of the composite part number as provided in the constructor
95  * to individual part numbers. depending on the format of the composite part number i.e.
96  * does it contain brackets and/or spaces etc.

99  private void convert(String partNoString)
100  {
101  if (partNoString.endsWith(")")==false &&
102  (partNoString.contains("(")==true && partNoString.contains(" ")==false ||
103  partNoString.contains("(")==true && partNoString.contains(" ")==true))
104  {bracketsNSpacesTrunc(partNoString);}
105  else if(partNoString.contains("(")==true && partNoString.contains(" ")==false ||
106  partNoString.contains("(")==true && partNoString.contains(" ")==true)
107  {bracketsNSpaces(partNoString);}
108  else if (partNoString.contains("(")==false && partNoString.contains(" ")==true)
109  {spacesNoBrackets(partNoString);}
110  else if (partNoString.contains("(")==false && partNoString.contains(" ")==false)
111  {noBracketsNoSpaces(partNoString);}
112  else{error(partNoString);}

116  /**
117  * convert a composite part number containing brackets and spaces i.e. there is more than
118  * one part number or a quantity larger than 1. uses regex W00t :)
119  * @param input the composite part number
120  */
121  private void bracketsNSpaces(String input)
122  {
123  //match anything that starts with a ( then contains any number of digits and ends with
124  //a closing )
125  String quantity = "\\([0-9]*\\)( ?)*+";
126  //match anything that starts with any number of digits case insensitive letters and
127  // forward slashes followed by the above.
128  String partNoAndQty = "([0-9a-zA-Z/]*)\\(([0-9]*)\\)( ?)*";
129 
130  //split the 'input' string using 'pattern'
131  this.partNos_M = input.split(quantity);
132  //in 'input' replace all 'pattern2' matches with group 2 matches of 'pattern2' plus a
133  //space, then split the resulting string at every space.
134  String[] tmp = input.replaceAll(partNoAndQty, "$2 ").split(" ");
135  this.partQty_M = new int[this.partNos_M.length];
136  for(int i = 0; i < tmp.length ; ++i)
137  {
138  try
139  {this.partQty_M[i] = Integer.parseInt(tmp[i]);}
140  catch(NumberFormatException nfe)
141  {this.partQty_M[i]=-8008135; this.partNos_M[i] = "ERROR: "+nfe.getMessage();}
142  }
143  }
